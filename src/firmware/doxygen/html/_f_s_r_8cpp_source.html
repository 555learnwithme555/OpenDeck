24 inline int16_t mapAnalog_int16(int16_t x, int16_t in_min, int16_t in_max, int16_t out_min, int16_t out_max)
25 {
26  return (x - in_min) * (out_max - out_min) / (in_max - in_min) + out_min;
27 };

29 inline int16_t calibratePressure(int16_t value, pressureType_t type)

60 void Analog::checkFSRvalue(uint8_t analogID, uint16_t pressure)
61 {
62  uint8_t calibratedPressure = calibratePressure(pressure, velocity);
63 
64  if (calibratedPressure > 0)
65  {
66  if (!getFsrPressed(analogID))
67  {
68  //sensor is really pressed
69  setFsrPressed(analogID, true);
70  uint8_t note = database.read(DB_BLOCK_ANALOG, analogMIDIidSection, analogID);
71  midi.sendNoteOn(note, calibratedPressure, database.read(DB_BLOCK_MIDI, midiChannelSection, noteChannel));
72  leds.noteToState(note, calibratedPressure, true);
73  if (sysEx.configurationEnabled())
74  {
75  if ((rTimeMs() - getLastCinfoMsgTime(DB_BLOCK_BUTTON)) > COMPONENT_INFO_TIMEOUT)
76  {
77  sysEx.startResponse();
78  sysEx.addToResponse(COMPONENT_ID_STRING);
79  sysEx.addToResponse(DB_BLOCK_ANALOG);
80  sysEx.addToResponse(analogID);
81  sysEx.sendResponse();
82  updateCinfoTime(DB_BLOCK_BUTTON);
83  }
84  }
85  }
86  }
87  else
88  {
89  if (getFsrPressed(analogID))
90  {
91  setFsrPressed(analogID, false);
92  uint8_t note = database.read(DB_BLOCK_ANALOG, analogMIDIidSection, analogID);
93  midi.sendNoteOff(note, 0, database.read(DB_BLOCK_MIDI, midiChannelSection, noteChannel));
94  leds.noteToState(note, 0, true);
95  if (sysEx.configurationEnabled())
96  {
97  if ((rTimeMs() - getLastCinfoMsgTime(DB_BLOCK_BUTTON)) > COMPONENT_INFO_TIMEOUT)
98  {
99  sysEx.startResponse();
100  sysEx.addToResponse(COMPONENT_ID_STRING);
101  sysEx.addToResponse(DB_BLOCK_ANALOG);
102  sysEx.addToResponse(analogID);
103  sysEx.sendResponse();
104  updateCinfoTime(DB_BLOCK_BUTTON);
105  }
106  }
107  }
108  }
109 
110  //update values
111  lastAnalogueValue[analogID] = pressure;
112 }
