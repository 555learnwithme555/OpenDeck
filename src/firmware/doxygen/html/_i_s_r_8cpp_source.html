63 inline void storeDigitalIn(uint8_t column, uint8_t bufferIndex)
64 {
65  uint8_t data = 0;
66  uint8_t dataReorder = 0;
67 
68  //make room for new data
69  inputBuffer[bufferIndex] <<= 8;
70 
71  //pulse latch pin
72  pulseLowToHigh(SR_LATCH_PORT, SR_LATCH_PIN);
73 
74  for (int i=0; i<8; i++)
75  {
76  data <<= 1;
77  data |= readPin(SR_DIN_PORT, SR_DIN_PIN);
78  //pulse clock pin
79  pulseHightToLow(SR_CLK_PORT, SR_CLK_PIN);
80  }
81 
82  //reorder data to match rows on PCB layout
83  for (int i=0; i<8; i++)
84  bitWrite(dataReorder, i, bitRead(data, dmRowBitArray[i]));
85 
86  inputBuffer[bufferIndex] |= (uint64_t)dataReorder;
87 }

109 inline void ledRowOn(uint8_t rowNumber, uint8_t intensity)
110 {
111  if (intensity == 255)
112  {
113  //max value, don't use pwm
114  #ifdef LED_INVERT
115  setLow(*(ledRowPortArray[rowNumber]), ledRowPinArray[rowNumber]);
116  #else
117  setHigh(*(ledRowPortArray[rowNumber]), ledRowPinArray[rowNumber]);
118  #endif
119  return;
120  }
121 
122  #ifdef LED_INVERT
123  intensity = 255 - intensity;
124  #endif
125 
126  switch (rowNumber)
127  {
128  //turn off pwm if intensity is max
129  case 0:
130  OCR1C = intensity;
131  TCCR1A |= (1<<COM1C1);
132  break;
133 
134  case 1:
135  OCR4D = intensity;
136  TCCR4C |= (1<<COM4D1);
137  break;
138 
139  case 2:
140  OCR1A = intensity;
141  TCCR1A |= (1<<COM1A1);
142  break;
143 
144  case 3:
145  OCR4A = intensity;
146  TCCR4A |= (1<<COM4A1);
147  break;
148 
149  case 4:
150  OCR3A = intensity;
151  TCCR3A |= (1<<COM3A1);
152  break;
153 
154  case 5:

160  break;

164 inline void checkLEDs()
165 {
166  if (blinkEnabled)
167  {
168  if (!blinkTimerCounter)
169  {
170  //change blinkBit state and write it into ledState variable if LED is in blink state
171  for (int i=0; i<MAX_NUMBER_OF_LEDS; i++)
172  {
173  if (bitRead(ledState[i], LED_BLINK_ON_BIT))
174  {
175  if (blinkState)
176  bitSet(ledState[i], LED_BLINK_STATE_BIT);
177  else
178  bitClear(ledState[i], LED_BLINK_STATE_BIT);
179  }
180  }
181 
182  blinkState = !blinkState;
183  }
184  }
185 
186  //if there is an active LED in current column, turn on LED row
187  //do fancy transitions here
188  for (int i=0; i<NUMBER_OF_LED_ROWS; i++)
189  {
190  uint8_t ledNumber = activeLEDcolumn+i*NUMBER_OF_LED_COLUMNS;
191  uint8_t ledStateSingle = bitRead(ledState[ledNumber], LED_ACTIVE_BIT) && (bitRead(ledState[ledNumber], LED_BLINK_ON_BIT) == bitRead(ledState[ledNumber], LED_BLINK_STATE_BIT));
192 
193  ledStateSingle *= (NUMBER_OF_LED_TRANSITIONS-1);
194 
195  //don't bother with pwm if it's disabled
196  if (!pwmSteps && ledStateSingle)
197  {
198  ledRowOn(i, 255);
199  }
200  else
201  {
202  if (ledTransitionScale[transitionCounter[ledNumber]])
203  ledRowOn(i, ledTransitionScale[transitionCounter[ledNumber]]);
204 
205  if (transitionCounter[ledNumber] != ledStateSingle)
206  {
207  //fade up
208  if (transitionCounter[ledNumber] < ledStateSingle)
209  {
210  transitionCounter[ledNumber] += pwmSteps;
211 
212  if (transitionCounter[ledNumber] > ledStateSingle)
213  transitionCounter[ledNumber] = ledStateSingle;
214  }
215  else
216  {
217  //fade down
218  transitionCounter[ledNumber] -= pwmSteps;
219 
220  if (transitionCounter[ledNumber] < 0)
221  transitionCounter[ledNumber] = 0;
222  }
223  }
224  }
225  }
