47 void MIDI::send(midiMessageType_t inType, uint8_t inData1, uint8_t inData2, uint8_t inChannel)
48 {
49  //test if channel is valid
50  if (inChannel >= MIDI_CHANNEL_OFF || inChannel == MIDI_CHANNEL_OMNI || inType < 0x80)
51  {
52  if (useRunningStatus && dinEnabled)
53  mRunningStatus_TX = midiMessageInvalidType;
54 
55  return; //don't send anything
56  }
57 
58  if (inType <= midiMessagePitchBend)
59  {
60  //channel messages
61  //protection: remove MSBs on data
62  inData1 &= 0x7f;
63  inData2 &= 0x7f;
64 
65  const uint8_t status = getStatus(inType, inChannel);
66 
67  if (dinEnabled)
68  {
69  if (useRunningStatus)
70  {
71  if (mRunningStatus_TX != status)
72  {
73  //new message, memorize and send header
74  mRunningStatus_TX = status;
75  sendUARTwriteCallback(mRunningStatus_TX);
76  }
77  }
78  else
79  {
80  //don't care about running status, send the status byte
81  sendUARTwriteCallback(status);
82  }
83 
84  //send data
85  sendUARTwriteCallback(inData1);
86 
87  if ((inType != midiMessageProgramChange) && (inType != midiMessageAfterTouchChannel))
88  {
89  sendUARTwriteCallback(inData2);
90  }
91  }
92 
93  if (usbEnabled)
94  {
95  uint8_t midiEvent = (uint8_t)inType >> 4;
96  uint8_t data1 = getStatus(inType, inChannel);
97 
98  USBMIDIpacket_t MIDIEvent = (USBMIDIpacket_t)
99  {
100  .Event = midiEvent,
101 
102  .Data1 = data1,
103  .Data2 = inData1,
104  .Data3 = inData2,
105  };
106 
107  sendUSBwriteCallback(MIDIEvent);
108  }
109  }
110  else if (inType >= midiMessageTuneRequest && inType <= midiMessageSystemReset)
111  {
112  sendRealTime(inType); //system real-time and 1 byte
113  }
114 }

206 {
207  if (dinEnabled && (sendUARTwriteCallback != NULL))
208  {
209  if (!inArrayContainsBoundaries)
210  sendUARTwriteCallback(0xf0);
211 
212  for (uint16_t i=0; i<inLength; ++i)
213  sendUARTwriteCallback(inArray[i]);
214 
215  if (!inArrayContainsBoundaries)
216  sendUARTwriteCallback(0xf7);
217 
218  if (useRunningStatus)

615 bool MIDI::parse(midiInterfaceType_t type)
616 {
617  if (type == dinInterface)
618  {
619  if (sendUARTreadCallback == NULL)
620  return false;
621 
622  int16_t data = sendUARTreadCallback();
623 
624  if (data == -1)
625  return false; //no data available
626 
627  const uint8_t extracted = data;
628 
629  if (dinPendingMessageIndex == 0)
630  {
631  //start a new pending message
632  mPendingMessage[0] = extracted;
633 
634  //check for running status first (din only)
635  if (isChannelMessage(getTypeFromStatusByte(mRunningStatus_RX)))
636  {
637  //only channel messages allow Running Status
638  //if the status byte is not received, prepend it to the pending message
639  if (extracted < 0x80)
640  {
641  mPendingMessage[0] = mRunningStatus_RX;
642  mPendingMessage[1] = extracted;
643  dinPendingMessageIndex = 1;
644  }
645 
646  //else: well, we received another status byte,
647  //so the running status does not apply here
648  //it will be updated upon completion of this message
649  }
650 
651  switch (getTypeFromStatusByte(mPendingMessage[0]))
652  {
653  //1 byte messages
654  case midiMessageStart:
655  case midiMessageContinue:
656  case midiMessageStop:
657  case midiMessageClock:
658  case midiMessageActiveSensing:
659  case midiMessageSystemReset:
660  case midiMessageTuneRequest:
661  //handle the message type directly here.
662  dinMessage.type = getTypeFromStatusByte(mPendingMessage[0]);
663  dinMessage.channel = 0;
664  dinMessage.data1 = 0;
665  dinMessage.data2 = 0;
666  dinMessage.valid = true;
667 
668  // \fix Running Status broken when receiving Clock messages.
669  // Do not reset all input attributes, Running Status must remain unchanged.
670  //resetInput();
671 
672  //we still need to reset these
673  dinPendingMessageIndex = 0;
674  dinPendingMessageExpectedLenght = 0;
675 
676  return true;
677  break;
678 
679  //2 bytes messages
680  case midiMessageProgramChange:
681  case midiMessageAfterTouchChannel:
682  case midiMessageTimeCodeQuarterFrame:
683  case midiMessageSongSelect:
684  dinPendingMessageExpectedLenght = 2;
685  break;
686 
687  //3 bytes messages
688  case midiMessageNoteOn:
689  case midiMessageNoteOff:
690  case midiMessageControlChange:
691  case midiMessagePitchBend:
692  case midiMessageAfterTouchPoly:
693  case midiMessageSongPosition:
694  dinPendingMessageExpectedLenght = 3;
695  break;
696 
697  case midiMessageSystemExclusive:
698  //the message can be any length between 3 and MIDI_SYSEX_ARRAY_SIZE
699  dinPendingMessageExpectedLenght = MIDI_SYSEX_ARRAY_SIZE;
700  mRunningStatus_RX = midiMessageInvalidType;
701  dinMessage.sysexArray[0] = midiMessageSystemExclusive;
702  break;
703 
704  case midiMessageInvalidType:
705  default:
706  //this is obviously wrong
707  //let's get the hell out'a here
708  resetInput();
709  return false;
710  break;
711  }
712 
713  if (dinPendingMessageIndex >= (dinPendingMessageExpectedLenght - 1))
714  {
715  //reception complete
716  dinMessage.type = getTypeFromStatusByte(mPendingMessage[0]);
717  dinMessage.channel = getChannelFromStatusByte(mPendingMessage[0]);
718  dinMessage.data1 = mPendingMessage[1];
719 
720  //save data2 only if applicable
721  if (dinPendingMessageExpectedLenght == 3)
722  dinMessage.data2 = mPendingMessage[2];
723  else dinMessage.data2 = 0;
724 
725  dinPendingMessageIndex = 0;
726  dinPendingMessageExpectedLenght = 0;
727  dinMessage.valid = true;
728  return true;
729  }
730  else
731  {
732  //waiting for more data
733  dinPendingMessageIndex++;
734  }
735 
736  if (use1byteParsing)
737  {
738  //message is not complete.
739  return false;
740  }
741  else
742  {
743  //call the parser recursively
744  //to parse the rest of the message.
745  return parse(dinInterface);
746  }
747  }
748  else
749  {
750  //first, test if this is a status byte
751  if (extracted >= 0x80)
752  {
753  //reception of status bytes in the middle of an uncompleted message
754  //are allowed only for interleaved Real Time message or EOX
755  switch (extracted)
756  {
757  case midiMessageClock:
758  case midiMessageStart:
759  case midiMessageContinue:
760  case midiMessageStop:
761  case midiMessageActiveSensing:
762  case midiMessageSystemReset:
763  //here we will have to extract the one-byte message,
764  //pass it to the structure for being read outside
765  //the MIDI class, and recompose the message it was
766  //interleaved into without killing the running status..
767  //this is done by leaving the pending message as is,
768  //it will be completed on next calls
769  dinMessage.type = (midiMessageType_t)extracted;
770  dinMessage.data1 = 0;
771  dinMessage.data2 = 0;
772  dinMessage.channel = 0;
773  dinMessage.valid = true;
774  return true;
775  break;
776 
777  //end of sysex
778  case 0xf7:
779  if (dinMessage.sysexArray[0] == midiMessageSystemExclusive)
780  {
781  //store the last byte (EOX)
782  dinMessage.sysexArray[dinPendingMessageIndex++] = 0xf7;
783  dinMessage.type = midiMessageSystemExclusive;
784 
785  //get length
786  dinMessage.data1 = dinPendingMessageIndex & 0xff; //LSB
787  dinMessage.data2 = dinPendingMessageIndex >> 8; //MSB
788  dinMessage.channel = 0;
789  dinMessage.valid = true;
790 
791  resetInput();
792  return true;
793  }
794  else
795  {
796  //error
797  resetInput();
798  return false;
799  }
800  break;
801 
802  default:
803  break;
804  }
805  }
806 
807  //add extracted data byte to pending message
808  if (mPendingMessage[0] == midiMessageSystemExclusive)
809  dinMessage.sysexArray[dinPendingMessageIndex] = extracted;
810  else
811  mPendingMessage[dinPendingMessageIndex] = extracted;
812 
813  //now we are going to check if we have reached the end of the message
814  if (dinPendingMessageIndex >= (dinPendingMessageExpectedLenght - 1))
815  {
816  //"FML" case: fall down here with an overflown SysEx..
817  //this means we received the last possible data byte that can fit the buffer
818  //if this happens, try increasing MIDI_SYSEX_ARRAY_SIZE
819  if (mPendingMessage[0] == midiMessageSystemExclusive)
820  {
821  resetInput();
822  return false;
823  }
824 
825  dinMessage.type = getTypeFromStatusByte(mPendingMessage[0]);
826 
827  if (isChannelMessage(dinMessage.type))
828  dinMessage.channel = getChannelFromStatusByte(mPendingMessage[0]);
829  else
830  dinMessage.channel = 0;
831 
832  dinMessage.data1 = mPendingMessage[1];
833 
834  //save data2 only if applicable
835  if (dinPendingMessageExpectedLenght == 3)
836  dinMessage.data2 = mPendingMessage[2];
837  else dinMessage.data2 = 0;
838 
839  //reset local variables
840  dinPendingMessageIndex = 0;
841  dinPendingMessageExpectedLenght = 0;
842 
843  dinMessage.valid = true;
844 
845  //activate running status (if enabled for the received type)
846  switch (dinMessage.type)
847  {
848  case midiMessageNoteOff:
849  case midiMessageNoteOn:
850  case midiMessageAfterTouchPoly:
851  case midiMessageControlChange:
852  case midiMessageProgramChange:
853  case midiMessageAfterTouchChannel:
854  case midiMessagePitchBend:
855  //running status enabled: store it from received message
856  mRunningStatus_RX = mPendingMessage[0];
857  break;
858 
859  default:
860  //no running status
861  mRunningStatus_RX = midiMessageInvalidType;
862  break;
863  }
864 
865  return true;
866  }
867  else
868  {
869  //update the index of the pending message
870  dinPendingMessageIndex++;
871 
872  if (use1byteParsing)
873  return false; //message is not complete.
874  else
875  return parse(dinInterface); //call the parser recursively to parse the rest of the message.
876  }
877  }
878  }
879  else if (type == usbInterface)
880  {
881  USBMIDIpacket_t USBMIDIpacket;
882 
883  if (!sendUSBreadCallback(USBMIDIpacket))
884  return false; //nothing received
885 
886  //we already have entire message here
887  //MIDIEvent.Event is CIN, see midi10.pdf
888  //shift cin four bytes left to get midiMessageType_t
889  uint8_t midiMessage = USBMIDIpacket.Event << 4;
890 
891  switch(midiMessage)
892  {
893  //1 byte messages
894  case sysCommon1byteCin:
895  if (USBMIDIpacket.Data1 != 0xF7)
896  {
897  //this isn't end of sysex, it's 1byte system common message
898 
899  //case midiMessageClock:
900  //case midiMessageStart:
901  //case midiMessageContinue:
902  //case midiMessageStop:
903  //case midiMessageActiveSensing:
904  //case midiMessageSystemReset:
905  usbMessage.type = (midiMessageType_t)USBMIDIpacket.Data1;
906  usbMessage.channel = 0;

982  usbMessage.sysexArray[sysExArrayLength] = USBMIDIpacket.Data1;

1312 void MIDI::thruFilter(uint8_t inChannel, midiInterfaceType_t type, midiFilterMode_t filterMode)
1313 {
1314  //if the feature is disabled, don't do anything.
1315  if (filterMode == THRU_OFF)
1316  return;
1317

1319  msg = type == dinInterface ? &dinMessage : &usbMessage;
1320 
1321  bool savedDINstate = dinEnabled;
1322  bool savedUSBstate = usbEnabled;
1323 
1324  switch(filterMode)
1325  {
1326  case THRU_FULL_DIN:
1327  case THRU_SAME_CHANNEL_DIN:
1328  case THRU_DIFF_CHANNEL_DIN:
1329  setUSBMIDIstate(false);

1331 
1332  case THRU_FULL_USB:
1333  case THRU_SAME_CHANNEL_USB:
1334  case THRU_DIFF_CHANNEL_USB:
1335  setDINMIDIstate(false);
1336  break;
1337

1339  return;
1340  }
1341

1343  if ((*msg).type >= midiMessageNoteOff && (*msg).type <= midiMessagePitchBend)
1344  {
1345  const bool filter_condition = (((*msg).channel == mInputChannel) ||
1346  (mInputChannel == MIDI_CHANNEL_OMNI));
1347 
1348  //now let's pass it to the output
1349  switch (filterMode)
1350  {
1351  case THRU_FULL_DIN:
1352  send((*msg).type, (*msg).data1, (*msg).data2, (*msg).channel);
1353  break;
1354 
1355  case THRU_FULL_USB:
1356  send((*msg).type, (*msg).data1, (*msg).data2, (*msg).channel);
1357  break;
1358 
1359  case THRU_SAME_CHANNEL_DIN:
1360  if (filter_condition)
1361  {
1362  send((*msg).type, (*msg).data1, (*msg).data2, (*msg).channel);
1363  }
1364  break;
1365 
1366  case THRU_SAME_CHANNEL_USB:
1367  if (filter_condition)
1368  {
1369  send((*msg).type, (*msg).data1, (*msg).data2, (*msg).channel);
1370  }
1371  break;
1372 
1373  case THRU_DIFF_CHANNEL_DIN:
1374  if (!filter_condition)
1375  {
1376  send((*msg).type, (*msg).data1, (*msg).data2, (*msg).channel);
1377  }
1378  break;
1379 
1380  case THRU_DIFF_CHANNEL_USB:
1381  if (!filter_condition)
1382  {
1383  send((*msg).type, (*msg).data1, (*msg).data2, (*msg).channel);
1384  }
1385  break;
1386 
1387  default:
1388  break;
1389  }
1390  }
1391  else
1392  {
1393  //always forward system messages
1394  switch ((*msg).type)
1395  {
1396  //real Time and 1 byte
1397  case midiMessageClock:
1398  case midiMessageStart:
1399  case midiMessageStop:
1400  case midiMessageContinue:
1401  case midiMessageActiveSensing:
1402  case midiMessageSystemReset:
1403  case midiMessageTuneRequest:
1404  sendRealTime((*msg).type);
1405  break;
1406

1408  //send SysEx (0xf0 and 0xf7 are included in the buffer)
1409  sendSysEx(getSysExArrayLength(type), getSysExArray(type), true);
1410  break;
1411

1413  sendSongSelect((*msg).data1);
1414  break;
1415 
1416  case midiMessageSongPosition:
1417  sendSongPosition((*msg).data1 | ((unsigned)(*msg).data2 << 7));
1418  break;
1419 
1420  case midiMessageTimeCodeQuarterFrame:
1421  sendTimeCodeQuarterFrame((*msg).data1, (*msg).data2);
1422  break;
1423 
1424  default:
1425  break;
1426  }
1427  }

1430  setDINMIDIstate(savedDINstate);
