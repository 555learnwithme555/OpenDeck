22 void Analog::checkPotentiometerValue(uint8_t analogID, uint16_t tempValue)
23 {
24  //calculate difference between current and previous reading
25  uint16_t analogDiff = abs(tempValue - lastAnalogueValue[analogID]);
26 
27  if (analogDiff < POTENTIOMETER_CC_STEP)
28  return;
29 
30  uint8_t ccValue = RAW_ADC_2_MIDI(tempValue);
31  uint8_t oldCCvalue = RAW_ADC_2_MIDI(lastAnalogueValue[analogID]);
32 
33  if (ccValue == oldCCvalue)
34  return;
35 
36  //invert CC data if potInverted is true
37  if (database.read(DB_BLOCK_ANALOG, analogInvertedSection, analogID))
38  ccValue = 127 - ccValue;
39

41  uint8_t upperCClimit = database.read(DB_BLOCK_ANALOG, analogCCupperLimitSection, analogID);
42 
43  //only use map when cc limits are different from defaults
44  if ((lowerCClimit != 0) || (upperCClimit != 127))
45  {
46  if (database.read(DB_BLOCK_ANALOG, analogTypeSection, analogID) == aType_potentiometer_cc)
47  midi.sendControlChange(database.read(DB_BLOCK_ANALOG, analogMIDIidSection, analogID), mapAnalog_uint8(ccValue, 0, 127, lowerCClimit, upperCClimit), database.read(DB_BLOCK_MIDI, midiChannelSection, CCchannel));
48  else
49  midi.sendNoteOn(database.read(DB_BLOCK_ANALOG, analogMIDIidSection, analogID), mapAnalog_uint8(ccValue, 0, 127, lowerCClimit, upperCClimit), database.read(DB_BLOCK_MIDI, midiChannelSection, CCchannel));
50 
51  if (sysEx.configurationEnabled())

57  sysEx.addToResponse(DB_BLOCK_ANALOG);

61  }
62  }

83  }
84 
85  //update values
