162 void mcuReset()
163 {
164  cli();
165  //stop watchdog timer, if running
166  MCUSR &= ~(1<<WDFR);
167  WDTCSR |= (1<<WDCE);
168  WDTCSR = 0;
169  _delay_ms(5);
170  disablePeripherals();
171 
172  wdt_enable(WDTO_15MS);
173  while(1);
174 }
