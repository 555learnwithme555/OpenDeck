165 void SysEx::decode()
166 {
167  decodedMessage.status = (sysExStatus_t)sysExArray[(uint8_t)statusByte];
168 
169  if (decodedMessage.status != REQUEST)
170  {
171  //don't let status be anything but request
172  setStatus(ERROR_STATUS);
173  return;
174  }
175 
176  if (!checkID())
177  return; //don't send response to wrong ID
178 
179  if (checkSpecialRequests())
180  return; //special request was handled by now
181 
182  //message appears to be fine for now
183  if (!sysExEnabled)
184  {
185  //message is fine, but handshake hasn't been received
186  setStatus(ERROR_HANDSHAKE);
187  return;
188  }
189 
190  if (sysExArraySize < REQUEST_SIZE)
191  {
192  setStatus(ERROR_MESSAGE_LENGTH);
193  return;
194  }
195 
196  //don't try to request these parameters if the size is too small
197  decodedMessage.part = sysExArray[(uint8_t)partByte];
198  decodedMessage.wish = (sysExWish)sysExArray[(uint8_t)wishByte];
199  decodedMessage.amount = (sysExAmount)sysExArray[(uint8_t)amountByte];
200  decodedMessage.block = sysExArray[(uint8_t)blockByte];
201  decodedMessage.section = sysExArray[(uint8_t)sectionByte];
202 
203  if (!checkRequest())
204  return;
205 
206  uint16_t minLength = generateMinMessageLenght();
207 
208  if (sysExArraySize < minLength || !minLength)
209  {
210  setStatus(ERROR_MESSAGE_LENGTH);
211  return;
212  }
213 
214  if (checkParameters())
215  {
216  //message is ok
217  }
218 }

241 bool SysEx::checkSpecialRequests()
242 {
243  if (sysExArraySize != MIN_MESSAGE_LENGTH)
244  return false;
245 
246  switch(sysExArray[wishByte])
247  {
248  case SYSEX_CLOSE_REQUEST:
249  if (!sysExEnabled)
250  {
251  //connection can't be closed if it isn't opened
252  setStatus(ERROR_HANDSHAKE);
253  return true;
254  }
255  else
256  {
257  //close sysex connection
258  sysExEnabled = false;
259  setStatus(ACK);
260  return true;
261  }
262  break;
263 
264  case HANDSHAKE_REQUEST:
265  //hello message, necessary for allowing configuration
266  sysExEnabled = true;
267  setStatus(ACK);
268  return true;
269 
270  case BYTES_PER_VALUE_REQUEST:
271  if (sysExEnabled)
272  {
273  setStatus(ACK);
274  addToResponse(PARAM_SIZE);
275  }
276  else
277  {
278  setStatus(ERROR_HANDSHAKE);
279  }
280  return true;
281 
282  case PARAMS_PER_MESSAGE_REQUEST:
283  if (sysExEnabled)
284  {
285  setStatus(ACK);
286  addToResponse(PARAMETERS_PER_MESSAGE);
287  }
288  else
289  {
290  setStatus(ERROR_HANDSHAKE);
291  }
292  return true;
293 
294  default:
295  //check for custom string
296  for (int i=0; i<MAX_CUSTOM_REQUESTS; i++)
297  {
298  if (customRequests[i] == INVALID_VALUE)
299  continue;
300 
301  if (customRequests[i] != sysExArray[wishByte])
302  continue;
303 
304  if (sysExEnabled)
305  {
306  setStatus(ACK);
307  sendCustomRequestCallback(customRequests[i]);
308  }
309  else
310  {
311  setStatus(ERROR_HANDSHAKE);
312  }
313 
314  return true;
315  }
316  //custom string not found
317  setStatus(ERROR_WISH);
318  return true;
319  }
320 }

365 bool SysEx::checkParameters()
366 {
367  //sysex request is fine
368  //start building response
369  setStatus(ACK);
370 
371  uint8_t loops = 1, responseSize_ = responseSize;
372  uint16_t startIndex = 0, endIndex = 1;
373  bool allParts = false;
374 
375  if (decodedMessage.amount == sysExAmount_single)
376  {
377  #if PARAM_SIZE == 2
378  encDec_14bit decoded;
379  //index
380  decoded.high = sysExArray[indexByte];
381  decoded.low = sysExArray[indexByte+1];
382  decodedMessage.index = decoded.decode14bit();
383  #elif PARAM_SIZE == 1
384  decodedMessage.index = sysExArray[indexByte];
385  #endif
386  decodedMessage.index += (PARAMETERS_PER_MESSAGE*decodedMessage.part);
387 
388  if (decodedMessage.wish == sysExWish_set)
389  {
390  //new value
391  #if PARAM_SIZE == 2
392  decoded.high = sysExArray[newValueByte_single];
393  decoded.low = sysExArray[newValueByte_single+1];
394  decodedMessage.newValue = decoded.decode14bit();
395  #elif PARAM_SIZE == 1
396  decodedMessage.newValue = sysExArray[newValueByte_single];
397  #endif
398  }
399  }
400 
401  if ((decodedMessage.wish == sysExWish_backup) || (decodedMessage.wish == sysExWish_get))
402  {
403  if (decodedMessage.part == 127)
404  {
405  loops = sysExMessage[decodedMessage.block].section[decodedMessage.section].parts;
406  forcedSend = true;
407  allParts = true;
408  }
409 
410  if (decodedMessage.wish == sysExWish_backup)
411  {
412  //don't overwrite anything if backup is requested
413  responseSize_ = sysExArraySize - 1;
414  //convert response to request
415  sysExArray[(uint8_t)statusByte] = REQUEST;
416  //now convert wish to set
417  sysExArray[(uint8_t)wishByte] = (uint8_t)sysExWish_set;
418  //decoded message wish needs to be set to get so that we can retrieve parameters
419  decodedMessage.wish = sysExWish_get;
420  }
421  }
422 
423  for (int j=0; j<loops; j++)
424  {
425  responseSize = responseSize_;
426 
427  if (forcedSend)
428  {
429  decodedMessage.part = j;
430  sysExArray[partByte] = j;
431  }
432 
433  if (decodedMessage.amount == sysExAmount_all)
434  {
435  startIndex = PARAMETERS_PER_MESSAGE*decodedMessage.part;
436  endIndex = startIndex + PARAMETERS_PER_MESSAGE;
437 
438  if ((sysExParameter_t)endIndex > sysExMessage[decodedMessage.block].section[decodedMessage.section].numberOfParameters)
439  endIndex = sysExMessage[decodedMessage.block].section[decodedMessage.section].numberOfParameters;
440  }
441 
442  for (uint16_t i=startIndex; i<endIndex; i++)
443  {
444  switch(decodedMessage.wish)
445  {
446  case sysExWish_get:
447  if (decodedMessage.amount == sysExAmount_single)
448  {
449  if (!checkParameterIndex())
450  {
451  setStatus(ERROR_INDEX);
452  return false;
453  }
454  else
455  {
456  addToResponse(sendGetCallback(decodedMessage.block, decodedMessage.section, decodedMessage.index));
457  }
458  }
459  else
460  {
461  //get all params - no index is specified
462  addToResponse(sendGetCallback(decodedMessage.block, decodedMessage.section, i));
463  }
464  break;
465 
466  case sysExWish_set:
467  if (decodedMessage.amount == sysExAmount_single)
468  {
469  if (!checkParameterIndex())
470  {
471  setStatus(ERROR_INDEX);
472  return false;
473  }
474 
475  if (!checkNewValue())
476  {
477  setStatus(ERROR_NEW_VALUE);
478  return false;
479  }
480 
481  if (sendSetCallback(decodedMessage.block, decodedMessage.section, decodedMessage.index, decodedMessage.newValue))
482  return true;
483  }
484  else
485  {
486  uint8_t arrayIndex = (i-startIndex);
487 
488  #if PARAM_SIZE == 2
489  arrayIndex *= sizeof(sysExParameter_t);
490  arrayIndex += newValueByte_all;
491  encDec_14bit decoded;
492  decoded.high = sysExArray[arrayIndex];
493  decoded.low = sysExArray[arrayIndex+1];
494  decodedMessage.newValue = decoded.decode14bit();
495  #elif PARAM_SIZE == 1
496  decodedMessage.newValue = sysExArray[arrayIndex+newValueByte_all];
497  #endif
498 
499  if (!checkNewValue())
500  {
501  setStatus(ERROR_NEW_VALUE);
502  return false;
503  }
504 
505  if (!sendSetCallback(decodedMessage.block, decodedMessage.section, i, decodedMessage.newValue))
506  {
507  setStatus(ERROR_WRITE);
508  return false;
509  }
510  }
511  break;
512 
513  default:
514  setStatus(ERROR_WISH);
515  return false;
516  }
517  }
518 
519  if (forcedSend)
520  {
521  sysExArray[responseSize] = 0xF7;
522  responseSize++;
523 
524  midi.sendSysEx(responseSize, sysExArray, true);
525  }
526  }
527 
528  if (allParts)
529  {
530  //send ACK message at the end
531  responseSize = 0;
532  addToResponse(0xF0);
533  addToResponse(SYS_EX_M_ID_0);
534  addToResponse(SYS_EX_M_ID_1);
535  addToResponse(SYS_EX_M_ID_2);
536  addToResponse(ACK);
537  addToResponse(0x7F);
538  addToResponse(decodedMessage.wish);
539  addToResponse(decodedMessage.amount);
540  addToResponse(decodedMessage.block);
541  addToResponse(decodedMessage.section);
542  addToResponse(0xF7);
543 
544  midi.sendSysEx(responseSize, sysExArray, true);
545  }
546 
547  return true;
548 }

554 uint8_t SysEx::generateMinMessageLenght()
555 {
556  uint16_t size = 0;
557 
558  switch(decodedMessage.amount)
559  {
560  case sysExAmount_single:
561  switch(decodedMessage.wish)
562  {
563  case sysExWish_get:
564  case sysExWish_backup:
565  return ML_REQ_STANDARD + sizeof(sysExParameter_t); //add parameter length
566  break;
567 
568  case sysExWish_set:
569  return ML_REQ_STANDARD + 2*sizeof(sysExParameter_t); //add parameter length and new value length
570  break;
571 
572  default:
573  return 0;
574  }
575  break;
576 
577  case sysExAmount_all:
578  switch(decodedMessage.wish)
579  {
580  case sysExWish_get:
581  case sysExWish_backup:
582  return ML_REQ_STANDARD;
583 
584  case sysExWish_set:
585  size = sysExMessage[decodedMessage.block].section[decodedMessage.section].numberOfParameters;
586 
587  if (size > PARAMETERS_PER_MESSAGE)
588  {
589  if ((decodedMessage.part+1) == sysExMessage[decodedMessage.block].section[decodedMessage.section].parts)
590  size = size - ((sysExMessage[decodedMessage.block].section[decodedMessage.section].parts-1)*PARAMETERS_PER_MESSAGE);
591  else
592  size = PARAMETERS_PER_MESSAGE;
593  }
594 
595  size *= sizeof(sysExParameter_t);
596  size += ML_REQ_STANDARD;
597  return size;
598 
599  default:
600  return 0;
601  }
602  break;
603 
604  default:
605  return 0;
606  }
607 
608  return size;
609 }

652 {

654  {
655  case sysExWish_get:
656  if (decodedMessage.part == 127)
657  return true;
658 
659  if (decodedMessage.part >= sysExMessage[decodedMessage.block].section[decodedMessage.section].parts)
660  return false;
661  else
662  return true;

664 
665  case sysExWish_set:
666  case sysExWish_backup:
667  if (decodedMessage.wish == sysExWish_backup)
668  {
669  if (decodedMessage.part == 127)
