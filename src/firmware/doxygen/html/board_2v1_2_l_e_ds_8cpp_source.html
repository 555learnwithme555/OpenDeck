36 uint8_t Board::getRGBaddress(uint8_t rgbID, rgbIndex_t index)
37 {
38  //get RGB LED address for specified index
39  uint8_t column = rgbID % NUMBER_OF_LED_COLUMNS;
40  uint8_t row = (rgbID/NUMBER_OF_BUTTON_COLUMNS)*3;
41  uint8_t address = column + NUMBER_OF_LED_COLUMNS*row;
42 
43  switch(index)
44  {
45  case rgb_R:
46  return address;
47 
48  case rgb_G:
49  return address + NUMBER_OF_LED_COLUMNS*1;
50 
51  case rgb_B:
52  return address + NUMBER_OF_LED_COLUMNS*2;
53  }
54 
55  return 0;
56 }

58 uint8_t Board::getRGBID(uint8_t ledNumber)
