28 void Analog::update()
29 {
30  if (!board.analogDataAvailable())
31  return;
32 
33  int16_t analogData;
34 
35  //check values
36  for (int i=0; i<MAX_NUMBER_OF_ANALOG; i++)
37  {
38  analogData = board.getAnalogValue(i);
39 
40  //don't process component if it's not enabled
41  if (!database.read(DB_BLOCK_ANALOG, analogEnabledSection, i))
42  continue;
43 
44  if (database.read(DB_BLOCK_ANALOG, analogTypeSection, i) != aType_button)
45  {
46  #ifdef ENABLE_HYSTERESIS
47  analogData = getHysteresisValue(i, analogData);
48  #endif
49  analogType_t type = (analogType_t)database.read(DB_BLOCK_ANALOG, analogTypeSection, i);
50 
51  switch(type)
52  {
53  case aType_potentiometer_cc:
54  case aType_potentiometer_note:
55  checkPotentiometerValue(i, analogData);
56  break;
57 
58  case aType_fsr:
59  checkFSRvalue(i, analogData);
60  break;
61 
62  default:
63  break;
64  }
65  }
66  else
67  {
68  //read raw value
69  analogData = board.getAnalogValue(i);
70 
71  if (analogData == -1)
72  {
73  continue;
74  }
75 
76  bool state = analogData > DIGITAL_VALUE_THRESHOLD;
77  buttons.processButton(i+MAX_NUMBER_OF_BUTTONS, state, false);
78  }
79  }
80 
81  board.continueADCreadout();
82 }

108 
109  if (value > 1023)
