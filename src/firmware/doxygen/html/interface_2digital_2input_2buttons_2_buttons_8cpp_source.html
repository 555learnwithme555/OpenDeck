44 void Buttons::processMomentaryButton(uint8_t buttonID, bool buttonState, buttonMIDImessage_t midiMessage)
45 {
46  if (buttonState)
47  {
48  //send note on only once
49  if (!getButtonPressed(buttonID))
50  {
51  uint8_t note = database.read(DB_BLOCK_BUTTON, buttonMIDIidSection, buttonID);
52  setButtonPressed(buttonID, true);
53 
54  switch(midiMessage)
55  {
56  case buttonPC:
57  midi.sendProgramChange(note, database.read(DB_BLOCK_MIDI, midiChannelSection, programChangeChannel));
58  break;
59 
60  case buttonNote:
61  midi.sendNoteOn(note, 127, database.read(DB_BLOCK_MIDI, midiChannelSection, noteChannel));
62  leds.noteToState(note, 127, true);
63  break;
64 
65  case buttonCC:
66  midi.sendControlChange(note, 127, database.read(DB_BLOCK_MIDI, midiChannelSection, noteChannel));
67  break;
68 
69  default:
70  break;
71  }
72 
73  if (sysEx.configurationEnabled())
74  {
75  if ((rTimeMs() - getLastCinfoMsgTime(DB_BLOCK_BUTTON)) > COMPONENT_INFO_TIMEOUT)
76  {
77  sysEx.startResponse();
78  sysEx.addToResponse(COMPONENT_ID_STRING);
79  sysEx.addToResponse(DB_BLOCK_BUTTON);
80  sysEx.addToResponse(buttonID);
81  sysEx.sendResponse();
82  updateCinfoTime(DB_BLOCK_BUTTON);
83  }
84  }
85  }
86  }
87  else
88  {
89  //button is released
90  if (getButtonPressed(buttonID))
91  {
92  uint8_t note = database.read(DB_BLOCK_BUTTON, buttonMIDIidSection, buttonID);
93 
94  switch(midiMessage)
95  {
96  case buttonPC:
97  //do nothing
98  break;
99 
100  case buttonNote:
101  midi.sendNoteOff(note, 0, database.read(DB_BLOCK_MIDI, midiChannelSection, noteChannel));
102  leds.noteToState(note, 0, true);
103  break;
104 
105  case buttonCC:
106  midi.sendControlChange(note, 0, database.read(DB_BLOCK_MIDI, midiChannelSection, noteChannel));
107  break;
108 
109  default:
110  break;
111  }
112 
113  if (sysEx.configurationEnabled())
114  {
115  if ((rTimeMs() - getLastCinfoMsgTime(DB_BLOCK_BUTTON)) > COMPONENT_INFO_TIMEOUT)
116  {
117  sysEx.startResponse();
118  sysEx.addToResponse(COMPONENT_ID_STRING);
119  sysEx.addToResponse(DB_BLOCK_BUTTON);
120  sysEx.addToResponse(buttonID);
121  sysEx.sendResponse();
122  updateCinfoTime(DB_BLOCK_BUTTON);
123  }
124  }
125 
126  setButtonPressed(buttonID, false);
127  }
128  }
129 }

131 void Buttons::processLatchingButton(uint8_t buttonID, bool buttonState, buttonMIDImessage_t midiMessage)
132 {
133  if (buttonState != getPreviousButtonState(buttonID))
134  {
135  uint8_t note = database.read(DB_BLOCK_BUTTON, buttonMIDIidSection, buttonID);
136  uint8_t channel = database.read(DB_BLOCK_MIDI, midiChannelSection, noteChannel);
137 
138  if (buttonState)
139  {
140  //button is pressed
141  //if a button has been already pressed
142  if (getButtonPressed(buttonID))
143  {
144  midi.sendNoteOff(note, 0, channel);
145  leds.noteToState(note, 0, true);
146  if (sysEx.configurationEnabled())
147  {
148  if ((rTimeMs() - getLastCinfoMsgTime(DB_BLOCK_BUTTON)) > COMPONENT_INFO_TIMEOUT)
149  {
150  sysEx.startResponse();
151  sysEx.addToResponse(COMPONENT_ID_STRING);
152  sysEx.addToResponse(DB_BLOCK_BUTTON);
153  sysEx.addToResponse(buttonID);
154  sysEx.sendResponse();
155  updateCinfoTime(DB_BLOCK_BUTTON);
156  }
157  }
158 
159  //reset pressed state
160  setButtonPressed(buttonID, false);
161  }
162  else
163  {
164  //send note on
165  midi.sendNoteOn(note, 127, channel);
166  leds.noteToState(note, 127, true);
167  if (sysEx.configurationEnabled())
168  {
169  if ((rTimeMs() - getLastCinfoMsgTime(DB_BLOCK_BUTTON)) > COMPONENT_INFO_TIMEOUT)
170  {
171  sysEx.startResponse();
172  sysEx.addToResponse(COMPONENT_ID_STRING);
173  sysEx.addToResponse(DB_BLOCK_BUTTON);
174  sysEx.addToResponse(buttonID);
175  sysEx.sendResponse();
176  updateCinfoTime(DB_BLOCK_BUTTON);
177  }
178  }
179 
180  //toggle buttonPressed flag to true
181  setButtonPressed(buttonID, true);
182  }
183  }
184  }

187 void Buttons::processButton(uint8_t buttonID, bool state, bool debounce)
188 {
189  bool debounced = debounce ? buttonDebounced(buttonID, state) : true;
190 
191  if (debounced)
192  {
193  buttonType_t type = (buttonType_t)database.read(DB_BLOCK_BUTTON, buttonTypeSection, buttonID);
194 
195  buttonMIDImessage_t midiMessage = (buttonMIDImessage_t)database.read(DB_BLOCK_BUTTON, buttonMIDImessageSection, buttonID);
196 
197  if (midiMessage == buttonPC)
198  {
199  //ignore momentary/latching modes if button sends program change
200  //when released, don't send anything
201  processMomentaryButton(buttonID, state, midiMessage);
202  }
203  else
204  {
205  switch (type)
206  {
207  case buttonMomentary:
208  processMomentaryButton(buttonID, state, midiMessage);
209  break;
210 
211  case buttonLatching:
212  processLatchingButton(buttonID, state, midiMessage);
213  break;
214 
215  default:
216  break;
217  }
218  }
219 
220  updateButtonState(buttonID, state);
221  }
222 }

224 void Buttons::update()
225 {
226  for (int i=0; i<MAX_NUMBER_OF_BUTTONS; i++)
227  {
228  bool buttonState;
229  uint8_t encoderPairIndex = board.getEncoderPair(i);
230 
231  if (database.read(DB_BLOCK_ENCODER, encoderEnabledSection, encoderPairIndex))
232  buttonState = false; //button is member of encoder pair, always set state to released
233  else
234  buttonState = board.getButtonState(i);
235 
236  processButton(i, buttonState);
237  }
238 }

240 void Buttons::updateButtonState(uint8_t buttonID, uint8_t buttonState)
241 {
242  uint8_t arrayIndex = buttonID/8;
243  uint8_t buttonIndex = buttonID - 8*arrayIndex;
244 
245  //update state if it's different than last one
246  if (BIT_READ(previousButtonState[arrayIndex], buttonIndex) != buttonState)
247  BIT_WRITE(previousButtonState[arrayIndex], buttonIndex, buttonState);

258 bool Buttons::buttonDebounced(uint8_t buttonID, bool buttonState)
259 {
260  //shift new button reading into previousButtonState
261  buttonDebounceCounter[buttonID] = (buttonDebounceCounter[buttonID] << 1) | buttonState | buttonDebounceCompare;
262 
263  //if button is debounced, return true
264  return ((buttonDebounceCounter[buttonID] == buttonDebounceCompare) || (buttonDebounceCounter[buttonID] == 0xFF));
265 }
