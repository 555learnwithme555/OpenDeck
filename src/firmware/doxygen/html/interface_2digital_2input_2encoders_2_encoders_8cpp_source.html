31  for (int i=0; i<MAX_NUMBER_OF_ENCODERS; i++)
32  {
33  if (!database.read(DB_BLOCK_ENCODER, encoderEnabledSection, i))
34  continue;
35 
36  encoderPosition_t encoderState = (encoderPosition_t)board.getEncoderState(i);
37  if (encoderState == encStopped)
38  continue;
39 
40  if (database.read(DB_BLOCK_ENCODER, encoderInvertedSection, i))
41  {
42  if (encoderState == encMoveLeft)
43  encoderState = encMoveRight;
44  else
45  encoderState = encMoveLeft;
46  }
47 
48  switch((encoderType_t)database.read(DB_BLOCK_ENCODER, encoderEncodingModeSection, i))
49  {
50  case enc7Fh01h:
51  if (encoderState == encMoveLeft)
52  encoderValue = ENCODER_VALUE_LEFT_7FH01H;
53  else
54  encoderValue = ENCODER_VALUE_RIGHT_7FH01H;
55  break;
56 
57  case enc3Fh41h:
58  if (encoderState == encMoveLeft)
59  encoderValue = ENCODER_VALUE_LEFT_3FH41H;
60  else
61  encoderValue = ENCODER_VALUE_RIGHT_3FH41H;
62  break;
63 
64  default:
65  continue;
66  break;
67  }
68 
69  midi.sendControlChange(database.read(DB_BLOCK_ENCODER, encoderMIDIidSection, i), encoderValue, database.read(DB_BLOCK_MIDI, midiChannelSection, CCchannel));

71  {
72  if ((rTimeMs() - getLastCinfoMsgTime(DB_BLOCK_ENCODER)) > COMPONENT_INFO_TIMEOUT)
