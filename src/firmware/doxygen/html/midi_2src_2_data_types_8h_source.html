141  void encodeTo14bit()
142  {
143  uint8_t newHigh = (value >> 8) & 0xFF;
144  uint8_t newLow = value & 0xFF;
145  newHigh = (newHigh << 1) & 0x7F;
146 
147  if ((newLow >> 7) & 0x01)
148  {
149  newHigh |= 0x01;
150  }
151  else
152  {
153  newHigh &= ~0x01;
154  }
155 
156  newLow = lowByte_7bit(newLow);
157  high = newHigh;
158  low = newLow;
159  }
160 
161  uint16_t decode14bit()
162  {
163  if (high & 0x01)
164  {
165  low |= (1 << 7);
166  }
167  else
168  {
169  low &= ~(1 << 7);
170  }
171 
172  high >>= 1;
173 
174  uint16_t joined;
175  joined = high;
176  joined <<= 8;
177  joined |= low;
178 
179  return joined;
180  }
